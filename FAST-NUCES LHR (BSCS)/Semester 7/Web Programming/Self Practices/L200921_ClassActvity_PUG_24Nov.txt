const express = require("express");
const path = require("path");

const app = express();
const port = "8000";

app.set("views", path.join(__dirname, "views"));
app.set("view engine", "pug");

app.get('/', (req,res)=> {
    res.render('index')
});

app.listen(port, ()=> {
    console.log('Listening to request on http://localhost:8000');
})
